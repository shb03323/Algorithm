class Solution {
    public static int solution(int n, int[] lost, int[] reserve) {
        int answer = 0;
        boolean[] tempReserve = new boolean[reserve.length];
        boolean[] tempLost = new boolean[lost.length];

        for(int i = 0; i < tempReserve.length; i++) {
            tempReserve[i] = true;
        }
        for(int i = 0; i < tempLost.length; i++) {
            tempLost[i] = true;
        }

        for(int i = 0; i < lost.length; i++) {
            for(int j = 0; j < reserve.length; j++) {
                if(lost[i] == reserve[j]) {
                    answer++;
                    tempLost[i] = false;
                    tempReserve[j] = false;
                    break;
                }
            }
        }
        for(int i = 0; i < lost.length; i++) {
            for(int j = 0; j < reserve.length; j++) {
                if(lost[i] - 1 == reserve[j] && tempReserve[j] && tempLost[i]) {
                    answer++;
                    tempReserve[j] = false;
                    tempLost[i] = false;
                }

                if(lost[i] + 1 == reserve[j] && tempReserve[j] && tempLost[i]) {
                    answer++;
                    tempReserve[j] = false;
                    tempLost[i] = false;
                }
            }
        }
        answer = n + answer - lost.length;
        return answer;
    }
}
